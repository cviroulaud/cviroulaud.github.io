\section{Problématique}
\begin{frame}
    \frametitle{}
    $$
        F_n = \left\{
        \begin{array}{ll}
            F_0 = 0                 & \mbox{si } n=0 \\
            F_1=1                   & \mbox{si } n=1 \\
            F_{n} = F_{n-1}+F_{n-2} & \mbox{si } n>1 \\
        \end{array}
        \right.
    $$
    \begin{center}
        \framebox{Comment obtenir un calcul efficace des termes de la suite?}
    \end{center}
