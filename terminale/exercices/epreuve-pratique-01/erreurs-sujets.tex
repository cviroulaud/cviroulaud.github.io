\begin{lstlisting}[language=Python  , xleftmargin=2em, xrightmargin=2em]
for i in range(len(image)):
    for j in range(...):
        if image[i][j] < ... :
\end{lstlisting}
et non
\begin{lstlisting}[language=Python  , xleftmargin=2em, xrightmargin=2em]
for i in range(len(image)):

\begin{lstlisting}[language=Python  , xleftmargin=2em, xrightmargin=2em]
